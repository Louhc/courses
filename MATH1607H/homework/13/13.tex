        \subsection*{1.(3)}
            $$
                S_m=\sum_{n=1}^{m}\frac{1}{n(n+1)(n+2)}=\sum_{n=1}^{m}(\frac{1}{2n(n+1)}-\frac{1}{2(n+1)(n+2)})=\frac14-\frac{1}{2(m+1)(m+2)}
            $$
            因此 $\{S_m\}$ 收敛.
            $$
                S=\lim_{m\to+\infty}S_m=\frac14
            $$
